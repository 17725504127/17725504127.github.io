# 变量

# var、let 和const的区别

# 一、var

  1. 存在变量预解析(预解析就是变量的声明会提前，但赋值不会提前)
console.log(a)//undefined
var a=1
console.log(a)//1
  1. 可以重复声明(重复声明会后面的覆盖前面的变量)
var a=1
var a=2
console.log(a)//2
  1. 没有块级作用域，只有函数作用域和全局作用域(块级作用域有时候会很方便)
function test(){
    var a=1
    console.log(a)//1
}
test()
console.log(a)//a is not defined

if(true){
    var b=2
    console.log(b)//2
}
console.log(b)//2
  1. 无法限制修改值(就是可以随便的修改赋的值)
var a=1
a=10
console.log(a)//10

# 二、let

  1. 不存在变量预解析(声明之前是无法使用的)
console.log(a)//Cannot access 'a' before initialization
let a=1
console.log(a)//1
  1. 不可以重复声明
let a=1
let a=2
console.log(a)//Identifier 'a' has already been declared
  1. 存在块级作用域(块级作用域就是有{}的)
if(true){
    let a=1
    console.log(a)//1
}
console.log(a)//a is not defined
  1. 无法限制修改值
let a=1
a=2
console.log(a)//2

# 三、const

  1. 不存在变量预解析
console.log(a)//Cannot access 'a' before initialization
const a=1
console.log(a)//1
  1. 不可以重复声明
const a=1
const a=2
console.log(a)//Identifier 'a' has already been declared
  1. 存在块级作用域
if(true){
    const=1
    console.log(a)//1
}
console.log(a)//a is not defined
  1. 无法修改值(因为定义的是常量，所以值无法修改，并且一旦声明必须赋值,要不会报错)
const a=1
a=2
console.log(a)//Assignment to constant variable.

const b
console.log(b)//Missing initializer in const declaration

# 举例：let和var的直观区别

<body>
    <button>1</button>
    <button>2</button>
    <button>3</button>
</body>

var btn=document.querySelectorAll("button")
//var
    for(var i=0;i<btn.length;i++){
        btn[i].onclick=function(){
            console.log(i)
        }
    }// 3 3 3

//let
    for(let i=0;i<btn.length;i++){
        btn[i].onclick=function(){
            console.log(i)
        }
    }//0 1 2
